<?php

function GetNumberOfK($data, $k) 

{

        $left=getLeft($data,$k);

        $right=getRight($data,$k);

        return $right-$left+1;

}

function getLeft($arr,$k){

        $left=0;

        $right=count($arr)-1;

        $mid=intval($left+($right-$left)/2);

        while($left<=$right){

                if($arr[$mid]>=$k){//关键

                        $right=$mid-1;

                }else{

                        $left=$mid+1;

                }   

                $mid=intval($left+($right-$left)/2);

        }   

        return $left;

}

function getRight($arr,$k){

        $left=0;

        $right=count($arr)-1;

        $mid=intval($left+($right-$left)/2);

        while($left<=$right){

                if($arr[$mid]<=$k){//关键

                        $left=$mid+1;

                }else{

                        $right=$mid-1;

                }   

                $mid=intval($left+($right-$left)/2);

        }   

        return $right;

}
