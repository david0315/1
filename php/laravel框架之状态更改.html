    //jqery
    $(document).on("click",".se",function () {
           id=$(this).attr("id");
           sex= $(this).attr("ss");
           str="";
           if (sex==0){
               str="男"
           }else{
               str="女"
           }

            that=$(this);
           $.ajax({
               url:"jdjg",
               type:"post",
               dataType:"json",
               data:{
                   id:id,
                   sex:str
               },
              success:function (data) { console.log(data);
           if (data=="男"){
               that.text("男")
               that.attr("ss",0)
           } else{
               that.text("女")
               that.attr("ss",1)
           }


              }
           })
    })
