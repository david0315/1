        /**
     * 过滤字符串中表情
     * @param $str string 昵称
     * @return string
     */
    public function filterEmoji($str) {
        $str = preg_replace_callback('/./u', function (array $match) {
            return strlen($match[0]) >= 4 ? '' : $match[0];
        }, $str);
        return $str;
    }
