    function decodeUnicode($str)
    {
    return preg_replace_callback('/\\\\u([0-9a-f]{4})/i',
    create_function(
    '$matches',
    'return mb_convert_encoding(pack("H*", $matches[1]), "UTF-8", "UCS-2BE");'
    ),
    $str);
    }


    $arr = array('name1':"中文",'name2':'abc12');
    $jsonstr = decodeUnicode(json_encode($arr));
