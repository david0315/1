    jquery注册点击事件

     

    代码展示

     

    <?php

		session_start();
		
		$goods_id = $_GET['goods_id'];

		$choice = $_GET['choice'];

		// var_dump($_GET);
		// die();
		//1.点击的是添加按钮
		$arr = $_SESSION['gwc'];
		if($choice == 'jia'){
			//数量加1，
			$arr[$goods_id]['num']+=1;
			
			
		}else{//2.点击的是删除按钮,前端限制了数量为1时不会传过来

			$arr[$goods_id]['num']-=1;
			
		}
		// 重新写进session。
		$_SESSION['gwc'] = $arr;
		
       
		// var_dump($_GET);
		// die();
        echo "OK";
		exit();
?>
